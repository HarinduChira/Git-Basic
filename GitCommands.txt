//git version
git --version

//initialize git and create repository
git init

//get status
git status

//change the branch to main
git ini -b main

//track files to staging area 
git add filename.ext
git add .

//to unstage
git rm --cached filename.ext

//Commit edits with a message
git commit -m "message"

//check logs
git log
git log --pretty=oneline

//check difference between previous updates
git diff
git diff --staged

//add git remote repo
git branch -M main
git remote add origin url

//upload files
git push -u origin main

//adding tags
git tag -a versionNumber
git tag -a versionNumber -m "message"

//pushing tags
git push origin versionNumber

//show tag
git show versionNumber

//create a branch
git checkout -b branchName
git switch 
git switch -

//view branches
git branch
git branch --all

//delete a branch
git branc -d branchName
